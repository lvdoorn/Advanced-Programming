\begin{grammar}
	<Expr> ::= Expr `,' Expr
	\alt Expr1
	
	<Expr1> ::= Number
	\alt String
	\alt `true'
	\alt `false'
	\alt `undefined'
	\alt Ident
	\alt Expr1 `+' Expr1
	\alt Expr1 `-' Expr1
	\alt Expr1 `*' Expr1
	\alt Expr1 `\%' Expr1
	\alt Expr1 `<' Expr1
	\alt Expr1 `===' Expr1
	\alt Ident `=' Expr1
	\alt Ident `(' Exprs `)'
	\alt `[' Exprs `]'
	\alt `[' ArrayFor `]'
	\alt `(' Expr `)'
	
	<Exprs> ::= $\varepsilon$
	\alt Expr1 CommaExprs	
	
	<CommaExprs> ::= $\varepsilon$
	\alt `,' Expr1 CommaExprs
	
	<ArrayFor> ::= `for' `(' Ident `of' Expr1 `)' ArrayCompr
	
	<ArrayIf> ::= `if' `(' Expr1 `)' ArrayCompr
	
	<ArrayCompr> ::= Expr1
	\alt ArrayFor
	\alt ArrayIf
	
	<Ident> ::= (see below)
	
	<Number> ::= (see below)
	
	<String> ::= (see below)
\end{grammar}
